def QR_eigensolver(A_in, tol=1e-14, N_max=int(1e6)):
    Ak = np.copy(A_in)
    n = len(Ak)
    Qk = np.eye(n)
    for i in range(N_max):
        Q, R = QR_dec(Ak)
        Ak_old = np.copy(Ak)
        Ak = R @ Q
        
        if np.allclose(Ak, Ak_old, rtol=tol):
            print(f'Tolerance reached at step {i+1}')
            eigenVal = np.diag(Ak)
            eigenVect = Qk
            return eigenVal, eigenVect    # We are interested in the columns
\end{elegantcode}
